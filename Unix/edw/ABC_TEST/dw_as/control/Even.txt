#!/bin/sh                                                                                                                                                                                                                             
                                                                                                                                                                                                                                      
echo "Enter to number to print Even numbers upto this number starting from begining"                                                                                                                                                  
                                                                                                                                                                                                                                      
read a                                                                                                                                                                                                                                
                                                                                                                                                                                                                                      
for ((i=1; i <= $a; ++i))                                                                                                                                                                                                             
do                                                                                                                                                                       
                                                                                                                                                                                                                                      
     if  !(( i % 2 ))                                                                                                                                                                                                                 
      then                                                                                                                                                                                                                            
         echo $i                                                                                                                                                                                                                      
      fi                                                                                                                                                                                                                              
done
